    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index(Request $request)
    {
         try
        {
            if (!$user = JWTAuth::toUser($request->token)) {
                return response()->json(['user_not_found'], 404);
            }

        }
        catch (TokenExpiredException $e)
        {
            return response()->json(['token_expired'], $e->getStatusCode());

        }
        catch (TokenInvalidException $e)
        {
            return response()->json(['token_invalid'], $e->getStatusCode());

        }
        catch (JWTException $e)
        {
            return response()->json(['token_missing'], $e->getStatusCode());
        }
        $token = $request->input('token');
        
        //var_dump($token);
        $user = JWTAuth::toUser($token);

        $wishListBooks = Wishlist::where('user_id', $user->id);
        $wishListBooks = $wishListBooks->get()->toArray();
        
        return response()->json(["success" => $wishListBooks]);
        //
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {
                 try
        {
            if (!$user = JWTAuth::toUser($request->token)) {
                return response()->json(['user_not_found'], 404);
            }

        }
        catch (TokenExpiredException $e)
        {
            return response()->json(['token_expired'], $e->getStatusCode());

        }
        catch (TokenInvalidException $e)
        {
            return response()->json(['token_invalid'], $e->getStatusCode());

        }
        catch (JWTException $e)
        {
            return response()->json(['token_missing'], $e->getStatusCode());
        }
        $token = $request->input('token');
        $user = JWTAuth::toUser($token);

        if (count(Wishlist::where(['isbn' => $request->isbn, 'user_id' => $user->id])->get()->toArray()) == 0)
            {
                $wishlistBook = new Wishlist;
                $wishlistBook->isbn = $request->isbn;
                $wishlistBook->user_id = $user->id;
                $wishlistBook->hidden = $request->hidden;
                $wishlistBook->save();
                return response()->json(['success' => 'Book has been added into the wishlist of the user']);
            }
        else
            {
                return response()->json(['error' => 'Book is already set for this user in his wishlist'], 400);
            }
        

        //
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function destroy(Request $request)
    {
        try
        {
            if (!$user = JWTAuth::toUser($request->token)) {
                return response()->json(['user_not_found'], 404);
            }

        }
        catch (TokenExpiredException $e)
        {
            return response()->json(['token_expired'], $e->getStatusCode());

        }
        catch (TokenInvalidException $e)
        {
            return response()->json(['token_invalid'], $e->getStatusCode());

        }
        catch (JWTException $e)
        {
            return response()->json(['token_missing'], $e->getStatusCode());
        }
        $token = $request->input('token');
        
        //var_dump($token);
        $user = JWTAuth::toUser($token);
        
        //var_dump($token);
        //$user = JWTAuth::toUser($token);

        $targetBook = Wishlist::where(['user_id' => $user->id, 'isbn' => $request->isbn])->first();
        if ($targetBook != null)
            {
                $targetBook->forceDelete();
                return response()->json(['success' => 'The book has been deleted from the wishlist']);
            }
        else
            {
                return response()->json(['error' => 'The book does not exist in the wishlist'], 400);
            }
